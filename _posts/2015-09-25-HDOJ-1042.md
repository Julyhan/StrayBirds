```
inline unsigned cantor(char* p)//range from 0 to (N-1)!
{
	unsigned result = 0;
	for (size_t i = 1; i < N; i++)
	{
		unsigned num = 0;
		for (size_t j = 0; j <= i; j++)
		{
			if (p[i] < p[j])// Eg. 21
			{
				num++;
			}
		}
		result += fac[i] * num;
	}
	return result;
}
```
